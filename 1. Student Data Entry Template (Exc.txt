3. SQL Table for Student Data
CREATE TABLE Students (
    StudentID INT PRIMARY KEY,
    Name VARCHAR(100),
    Age INT,
    Gender VARCHAR(10),
    Course VARCHAR(50),
    Email VARCHAR(100),
    Phone VARCHAR(15),
    Address TEXT
);
